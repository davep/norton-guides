[]               Memory reference                                    Operator

  [expression]                      ; Ideal mode

    Specifies a memory reference.


  [expression]                      ; MASM mode
  expression1[expression2]
  [expression1][expression2]

    Specifies a memory reference
           or addition
           or register-indirect memory operands


    Examples:   Ideal
                mov     al,[bx]
                cmp     [word ptr es:bx],0a0dh
                mov     ax,[bx + si + 2]
                mov     dx,[table2 + bx]
                les     di,[bp + 8 + si + 4]
                call    [codeptr ds:bp + 6]
                lods    [byte es:si]            ; lodsb es:
                mov     [word low  old_i1bh],bx
                mov     [word high old_i1bh],es
