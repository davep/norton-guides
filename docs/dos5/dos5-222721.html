struc Arena
 arenaSignature db ?            ; 4Dh valid item, 5Ah last item
 arenaOwner     dw ?            ; Owner of arena item
 arenaSize      dw ?            ; Size of item, in paragraphs
 arenaReserved  db 3 dup (?)    ; Reserved
 arenaName      db 8 dup (?)    ; Owner filename (ASCIIZ if length < 8)
ends                            ; = 16d bytes

        ; Arena structures are created and maintained by DOS and
        ; located 1 paragraph before the allocated memory block.
        ; Programs may read, but not modify Arenas.
