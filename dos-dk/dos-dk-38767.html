   Flg. CONFIG.SYS og AUTOEXEC.BAT vil være typiske på systemer med DOS
   3.30, og EGA/VGA skærm:

   CONFIG.SYS


        COUNTRY=045,865,C:\DOS\COUNTRY.SYS
        DEVICE=C:\DOS\DISPLAY.SYS CON = (EGA,865)
        DEVICE=C:\MUS\MSMOUSE.SYS
        BUFFERS=20
        FILES=20
        BREAK=ON
        SHELL=C:\DOS\COMMAND.COM C:\DOS /P

   AUTOEXEC.BAT

        @ECHO OFF
        PROMPT $P$G
        PATH=C:\;C;\DOS
        MODE CON CODEPAGE PREPARE = ((865) C:\DOS\EGA.CPI)
        MODE CON CODEPAGE SELECT = 865
        C:\DOS\KEYB DK 865 C:\DOS\KEYBOARD.SYS
